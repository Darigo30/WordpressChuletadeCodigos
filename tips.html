function add_async_to_script($tag, $handle) {
    // agregar los handles o identificadores de los scripts al array
    $scripts_to_async = array('xxxx', 'xxxx'); //wp_register_script('xxxx', 'nombre-js.js', '', '', true);

    // recorremos el array y agregamos el atributo async defer
    foreach($scripts_to_async as $async_script) {

// Filter posts by taxonomy in admin
add_filter('parse_query', 'tsm_convert_id_to_term_in_query');
function tsm_convert_id_to_term_in_query($query) {
    global $pagenow;
    $post_type = 'XXXX'; // Cambiar al nombre del custom post type
    $taxonomy = 'XXXX'; // Cambiar al nombre del custom taxonomy
    $q_vars = &$query->query_vars;
    if ( $pagenow == 'edit.php' && isset($q_vars['post_type']) && $q_vars['post_type'] == $post_type && isset($q_vars[$taxonomy]) && is_numeric($q_vars[$taxonomy]) && $q_vars[$taxonomy] != 0 ) {
        $term = get_term_by('id', $q_vars[$taxonomy], $taxonomy);
        $q_vars[$taxonomy] = $term->slug;
    }
}
